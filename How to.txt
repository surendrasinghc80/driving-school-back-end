6. Create model
npx sequelize-cli model:generate --name Course --attributes name:string,slug:string,duration:string,fees:integer,description:string

7. Run Migrations
npm run db:migrate

8. Undo last migration:
npm run db:migrate:undo

9. Undo all migrations:
npm run db:migrate:undo:all

10. Generate a seeder:
npx sequelize-cli seed:generate --name demo-user

11. Run the seeder:
npm run db:seed


12. Undo seed:
npm run db:seed:undo

--------------------------------------------------------------------

9. Altering Tables
If you need to add a column:
npx sequelize-cli migration:generate --name add-age-to-user

Edit migration file:
export async function up(queryInterface, Sequelize) {
  await queryInterface.addColumn('Users', 'age', {
    type: Sequelize.INTEGER,
    allowNull: true
  });
}

export async function down(queryInterface, Sequelize) {
  await queryInterface.removeColumn('Users', 'age');
}


Run:
npm run db:migrate
